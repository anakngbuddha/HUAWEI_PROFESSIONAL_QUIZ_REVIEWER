  const handleAnswer = (questionIndex, answerIndex) => {
    const question = questions[questionIndex];
    
    if (question.type === 'multiple-answer') {
      const currentAnswers = selectedAnswers[questionIndex] || [];
      const newAnswers = currentAnswers.includes(answerIndex)
        ? currentAnswers.filter(a => a !== answerIndex)
        : [...currentAnswers, answerIndex];
      
      setSelectedAnswers({
        ...selectedAnswers,
        [questionIndex]: newAnswers
      });
    } else {
      setSelectedAnswers({
        ...selectedAnswers,
        [questionIndex]: answerIndex
      });
    }
  };

  const checkAnswer = (questionIndex) => {
    const question = questions[questionIndex];
    const userAnswer = selectedAnswers[questionIndex];

    if (userAnswer === undefined || (Array.isArray(userAnswer) && userAnswer.length === 0)) {
      return null;
    }

    let isCorrect = false;

    if (question.type === 'multiple-answer') {
      const sortedUser = [...userAnswer].sort();
      const sortedCorrect = [...question.correctAnswers].sort();
      isCorrect = JSON.stringify(sortedUser) === JSON.stringify(sortedCorrect);
    } else if (question.type === 'true-false') {
      isCorrect = userAnswer === question.correctAnswer;
    } else {
      isCorrect = userAnswer === question.correctAnswer;
    }

    if (!showExplanation[questionIndex] && isCorrect) {
      setScore(score + 1);
    }

    setShowExplanation({
      ...showExplanation,
      [questionIndex]: true
    });

    return isCorrect;

                        : 'border-red-500 bg-red-50'
                      : 'border-blue-500 bg-blue-50'
                    : isAnswered && optionIndex === question.correctAnswer

                    : 'border-gray-200 hover:border-gray-300'
                }`}
              >
                <input
                  type={question.type === 'multiple-answer' ? 'checkbox' : 'radio'}
                  name={`question-${index}`}
                  checked={
                    question.type === 'multiple-answer'
                      ? (userAnswer || []).includes(optionIndex)
                      : userAnswer === optionIndex
                  }
                  onChange={() => handleAnswer(index, optionIndex)}
                  disabled={isAnswered}
                  className="mr-3"
                />
                <span>{option}</span>
              </label>
            ))
          )}
        </div>

        {!isAnswered && userAnswer !== undefined && (question.type !== 'multiple-answer' || userAnswer.length > 0) && (

        )}

              {isCorrect ? (

    const question = questions[questionIndex];
    const userAnswer = selectedAnswers[questionIndex];

    if (userAnswer === undefined) return null;

    if (question.type === 'multiple-answer') {
      const sortedUser = [...userAnswer].sort();
      const sortedCorrect = [...question.correctAnswers].sort();
      return JSON.stringify(sortedUser) === JSON.stringify(sortedCorrect);
    } else if (question.type === 'true-false') {
      return userAnswer === question.correctAnswer;
    } else {
      return userAnswer === question.correctAnswer;
    }
  };

  const calculateScore = () => {
    let correctCount = 0;
    questions.forEach((_, index) => {
      if (showExplanation[index] && checkAnswerSilent(index)) {
        correctCount++;
      }
    });
    return correctCount;
  };

  const totalAnswered = Object.keys(showExplanation).length;
  const currentScore = calculateScore();

  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 p-6">
      <div className="max-w-4xl mx-auto">
        <div className="bg-white rounded-xl shadow-lg p-8 mb-6">

            </div>
            <div className="px-4 py-2 bg-green-100 rounded-lg">

            </div>
          </div>
        </div>

          {questions.map((question, index) => renderQuestion(question, index))}
        </div>

          <div className="bg-white rounded-xl shadow-lg p-8 mt-6 text-center">
            <h2 className="text-2xl font-bold text-gray-800 mb-4">Quiz Complete! 🎉</h2>
            <p className="text-xl mb-2">
              Your Final Score: <span className="font-bold text-blue-600">{currentScore} / {questions.length}</span>
            </p>
            <p className="text-gray-600">
              ({((currentScore / questions.length) * 100).toFixed(1)}%)
            </p>
          </div>
        )}
      </div>
    </div>
  );
};
